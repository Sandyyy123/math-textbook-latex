\section{Counting Objects}

\begin{learningobjective}
Students will be able to count objects up to 10 and write the corresponding numeral.
\end{learningobjective}

\begin{workedexample}
Count the apples and write the number.\\[8pt]
\textbf{🍎 🍎 🍎 🍎 🍎}\\[6pt]
There are \textbf{5} apples.
\end{workedexample}

\begin{remember}
Always count each object once. Point to each one as you count!
\end{remember}
