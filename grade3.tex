\section{Understanding Multiplication}

\begin{learningobjective}
Students will understand multiplication as repeated addition and use the times table for 2, 5, and 10.
\end{learningobjective}

\begin{workedexample}
Calculate $4 \times 3$.\\[6pt]
$4 \times 3$ means \textbf{4 groups of 3}:\\[4pt]
$3 + 3 + 3 + 3 = 12$\\[4pt]
\textbf{Answer: $4 \times 3 = 12$}
\end{workedexample}

\begin{remember}
$a \times b = b \times a$ \quad (the order does not change the answer)
\end{remember}
