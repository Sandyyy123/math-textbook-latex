\section{Adding Two-Digit Numbers}

\begin{learningobjective}
Students will add two-digit numbers, including cases that require regrouping (carrying tens).
\end{learningobjective}

\begin{workedexample}[Worked Example — Regrouping]
Calculate $24 + 38$.\\[6pt]
\begin{tabular}{r}
  $24$\\
  $+\;38$\\
  \hline
  $62$\\
\end{tabular}
\quad
\begin{minipage}{0.55\textwidth}
\small
Step 1: $4 + 8 = 12$. Write 2, carry 1.\\
Step 2: $2 + 3 + 1\text{(carried)} = 6$. Write 6.\\
\textbf{Answer: 62}
\end{minipage}
\end{workedexample}

\begin{remember}
Always add the ones column first, then the tens column.
\end{remember}

\section{Subtracting Two-Digit Numbers}
%% ────────────────────────────────────────────────────────

\begin{learningobjective}
Students will subtract two-digit numbers, including borrowing from the tens column.
\end{learningobjective}

\begin{workedexample}
Calculate $52 - 27$.\\[6pt]
\begin{tabular}{r}
  $52$\\
  $-\;27$\\
  \hline
  $25$\\
\end{tabular}
\quad
\begin{minipage}{0.55\textwidth}
\small
Step 1: $2 < 7$, so borrow from tens. $12 - 7 = 5$.\\
Step 2: $4 - 2 = 2$.\\
\textbf{Answer: 25}
\end{minipage}
\end{workedexample}
